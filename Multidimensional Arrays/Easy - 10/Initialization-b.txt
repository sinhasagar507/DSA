/*  Dynamic implementation of 2-D arrays in CPP - An array of vectors */

#include<bits/stdc++.h>
using namespace std;

int main()
{
   int m=3, n=2;
   
   int *arr[m];
   for(int i=0; i<m; i++)
        arr[i]=new int[n];
      
      
   for(int i=0; i<m; i++)
        for(int j=0; j<n; j++)
            cin>>arr[i][j];
            
            
    cout<<endl;
            
    for(int i=0; i<m; i++)
    {
        for(int j=0; j<n; j++)
            cout<<arr[i][j]<<" ";
            
        cout<<endl;
    }
}
